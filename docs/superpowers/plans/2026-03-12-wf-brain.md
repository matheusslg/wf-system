# wf-brain Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a persistent, searchable knowledge layer ("brain") for wf-system that gives AI agents long-lived project memory with semantic search.

**Architecture:** CLI tool (`scripts/wf-brain.js`) backed by SQLite + sqlite-vec for hybrid vector/keyword search, with a thin MCP server wrapper. Dependencies self-install from `.claude/mcp-servers/wf-brain/package.json`. Orchestrator and existing commands modified to auto-inject brain knowledge.

**Tech Stack:** Node.js, better-sqlite3, sqlite-vec, @xenova/transformers (MiniLM-L6-v2), @modelcontextprotocol/sdk

**Spec:** `docs/superpowers/specs/2026-03-12-wf-brain-design.md`

```js
'use strict';

const path = require('path');
const fs = require('fs');

// Resolve deps from MCP server's node_modules
const MODULE_PATH = path.join(__dirname, '..', '..', '.claude', 'mcp-servers', 'wf-brain', 'node_modules');
if (fs.existsSync(MODULE_PATH)) {
  module.paths.unshift(MODULE_PATH);
}

const Database = require('better-sqlite3');

const CURRENT_SCHEMA_VERSION = '1';

function initDb(dbPath) {
  const dir = path.dirname(dbPath);
  if (!fs.existsSync(dir)) {
    fs.mkdirSync(dir, { recursive: true });
  }

  const conn = new Database(dbPath);
  conn.pragma('journal_mode = WAL');
  conn.pragma('busy_timeout = 5000');

  // Check existing version
  const hasSchema = conn.prepare(
    "SELECT name FROM sqlite_master WHERE type='table' AND name='schema_meta'"
  ).get();

  if (!hasSchema) {
    // Fresh install
    conn.exec(`
      CREATE TABLE schema_meta (
        key TEXT PRIMARY KEY,
        value TEXT
      );

      CREATE TABLE entries (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        content TEXT NOT NULL,
        category TEXT NOT NULL,
        tags TEXT,
        source TEXT,
        embedding BLOB,
        created_at TEXT DEFAULT (datetime('now')),
        updated_at TEXT DEFAULT (datetime('now'))
      );

      CREATE TABLE pending (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        content TEXT NOT NULL,
        category TEXT NOT NULL,
        tags TEXT,
        source TEXT,
        proposed_by TEXT,
        embedding BLOB,
        created_at TEXT DEFAULT (datetime('now')),
        status TEXT DEFAULT 'pending'
      );

      INSERT INTO schema_meta (key, value) VALUES ('version', '${CURRENT_SCHEMA_VERSION}');
    `);

    // Try to load sqlite-vec for vector tables
    _initVecTables(conn);
  } else {
    const row = conn.prepare("SELECT value FROM schema_meta WHERE key='version'").get();
    const version = row ? row.value : '0';
    if (version < CURRENT_SCHEMA_VERSION) {
      _migrate(conn, version);
    }
  }

  return conn;
}

function _initVecTables(conn) {
  try {
    const sqliteVec = require('sqlite-vec');
    sqliteVec.load(conn);
    conn.exec(`
      CREATE VIRTUAL TABLE IF NOT EXISTS entries_vec USING vec0(
        id INTEGER PRIMARY KEY,
        embedding float[384]
      );
      CREATE VIRTUAL TABLE IF NOT EXISTS pending_vec USING vec0(
        id INTEGER PRIMARY KEY,
        embedding float[384]
      );
    `);
  } catch {
    // sqlite-vec not available — vector search disabled, keyword-only fallback
  }
}

function _migrate(conn, fromVersion) {
  // Future migrations go here: if (fromVersion < '2') { ... }
  conn.prepare("UPDATE schema_meta SET value = ? WHERE key = 'version'")
    .run(CURRENT_SCHEMA_VERSION);
}

function hasVec(conn) {
  try {
    conn.prepare("SELECT * FROM entries_vec LIMIT 0").get();
    return true;
  } catch {
    return false;
  }
}

// ---- CRUD: entries ----

function insertEntry(conn, { content, category, tags, source, embedding }) {
  const result = conn.prepare(
    'INSERT INTO entries (content, category, tags, source, embedding) VALUES (?, ?, ?, ?, ?)'
  ).run(content, category, tags || '', source || 'manual', embedding || null);
  const id = Number(result.lastInsertRowid);

  if (embedding && hasVec(conn)) {
    conn.prepare('INSERT INTO entries_vec (id, embedding) VALUES (?, ?)').run(id, embedding);
  }
  return id;
}

function getEntry(conn, id) {
  return conn.prepare('SELECT * FROM entries WHERE id = ?').get(id);
}

const ALLOWED_UPDATE_FIELDS = new Set(['content', 'category', 'tags', 'source', 'embedding']);

function updateEntry(conn, id, fields) {
  const sets = [];
  const vals = [];
  for (const [k, v] of Object.entries(fields)) {
    if (!ALLOWED_UPDATE_FIELDS.has(k)) continue;
    sets.push(`${k} = ?`);
    vals.push(v);
  }
  sets.push("updated_at = datetime('now')");
  vals.push(id);
  conn.prepare(`UPDATE entries SET ${sets.join(', ')} WHERE id = ?`).run(...vals);

  // Update vector if embedding changed
  if (fields.embedding && hasVec(conn)) {
    conn.prepare('DELETE FROM entries_vec WHERE id = ?').run(id);
    conn.prepare('INSERT INTO entries_vec (id, embedding) VALUES (?, ?)').run(id, fields.embedding);
  }
}

function deleteEntry(conn, id) {
  conn.prepare('DELETE FROM entries WHERE id = ?').run(id);
  if (hasVec(conn)) {
    try { conn.prepare('DELETE FROM entries_vec WHERE id = ?').run(id); } catch {}
  }
}

function listEntries(conn, { category, limit, recent } = {}) {
  let sql = 'SELECT * FROM entries';
  const params = [];
  if (category) {
    sql += ' WHERE category = ?';
    params.push(category);
  }
  sql += recent ? ' ORDER BY created_at DESC' : ' ORDER BY id DESC';
  sql += ` LIMIT ?`;
  params.push(limit || 20);
  return conn.prepare(sql).all(...params);
}

// ---- CRUD: pending ----

function insertPending(conn, { content, category, tags, source, proposedBy, embedding }) {
  const result = conn.prepare(
    'INSERT INTO pending (content, category, tags, source, proposed_by, embedding) VALUES (?, ?, ?, ?, ?, ?)'
  ).run(content, category, tags || '', source || '', proposedBy || '', embedding || null);
  const id = Number(result.lastInsertRowid);

  if (embedding && hasVec(conn)) {
    conn.prepare('INSERT INTO pending_vec (id, embedding) VALUES (?, ?)').run(id, embedding);
  }
  return id;
}

function getPending(conn, id) {
  return conn.prepare('SELECT * FROM pending WHERE id = ?').get(id);
}

function listPending(conn) {
  return conn.prepare("SELECT * FROM pending WHERE status = 'pending' ORDER BY id").all();
}

function approvePending(conn, id) {
  const pending = getPending(conn, id);
  if (!pending) throw new Error(`Pending entry ${id} not found`);

  const entryId = insertEntry(conn, {
    content: pending.content,
    category: pending.category,
    tags: pending.tags,
    source: pending.source,
    embedding: pending.embedding
  });

  conn.prepare("UPDATE pending SET status = 'approved' WHERE id = ?").run(id);
  return entryId;
}

function rejectPending(conn, id) {
  conn.prepare("UPDATE pending SET status = 'rejected' WHERE id = ?").run(id);
}

function approveAllPending(conn) {
  const pendings = listPending(conn);
  const ids = [];
  for (const p of pendings) {
    ids.push(approvePending(conn, p.id));
  }
  return ids;
}

// ---- Stats ----

function getStats(conn) {
  const totalEntries = conn.prepare('SELECT COUNT(*) as count FROM entries').get().count;
  const totalPending = conn.prepare("SELECT COUNT(*) as count FROM pending WHERE status = 'pending'").get().count;
  const cats = conn.prepare('SELECT category, COUNT(*) as count FROM entries GROUP BY category').all();
  const byCategory = {};
  for (const c of cats) byCategory[c.category] = c.count;
  return { totalEntries, totalPending, byCategory };
}

// ---- DB discovery ----

function findBrainDb(startDir) {
  let dir = path.resolve(startDir || process.cwd());
  for (let i = 0; i < 10; i++) {
    const candidate = path.join(dir, '.claude', 'brain.db');
    if (fs.existsSync(candidate)) return candidate;
    const parent = path.dirname(dir);
    if (parent === dir) break;
    dir = parent;
  }
  return null;
}

module.exports = {
  initDb, hasVec, findBrainDb,
  insertEntry, getEntry, updateEntry, deleteEntry, listEntries,
  insertPending, getPending, listPending, approvePending, rejectPending, approveAllPending,
  getStats
};
```

```js
'use strict';

const path = require('path');
const fs = require('fs');

const MODULE_PATH = path.join(__dirname, '..', '..', '.claude', 'mcp-servers', 'wf-brain', 'node_modules');
if (fs.existsSync(MODULE_PATH)) {
  module.paths.unshift(MODULE_PATH);
}

let _pipeline = null;

async function _getEmbedder() {
  if (_pipeline) return _pipeline;
  const { pipeline } = require('@xenova/transformers');
  _pipeline = await pipeline('feature-extraction', 'Xenova/all-MiniLM-L6-v2');
  return _pipeline;
}

async function generateEmbedding(text) {
  try {
    if (!text) return null;
    const embedder = await _getEmbedder();
    const output = await embedder(text);
    const float32 = new Float32Array(output.data);
    return Buffer.from(float32.buffer);
  } catch {
    return null;
  }
}

module.exports = { generateEmbedding };
```

```js
'use strict';

const db = require('./db');

/**
 * Compute keyword bonus: fraction of query words found in content.
 * Normalized to 0–0.2 range.
 */
function computeKeywordBonus(query, content) {
  if (!query || !content) return 0;
  const queryWords = query.toLowerCase().split(/\s+/).filter(w => w.length > 2);
  if (queryWords.length === 0) return 0;
  const contentLower = content.toLowerCase();
  let hits = 0;
  for (const word of queryWords) {
    if (contentLower.includes(word)) hits++;
  }
  return Math.min((hits / queryWords.length) * 0.2, 0.2);
}

/**
 * Keyword-only search fallback (when vector search is unavailable).
 * Scores entries by keyword overlap.
 */
function keywordSearch(conn, query, { limit = 5, category } = {}) {
  if (!query || query.trim() === '') {
    // Empty query: return most recent entries
    return db.listEntries(conn, { category, limit, recent: true })
      .map(e => ({ ...e, matchPercent: 100 }));
  }

  let entries = db.listEntries(conn, { category, limit: 1000 });
  const scored = entries.map(entry => {
    const bonus = computeKeywordBonus(query, entry.content + ' ' + (entry.tags || ''));
    // For keyword-only, normalize bonus from 0-0.2 range to 0-100
    const matchPercent = Math.round(bonus * 500);
    return { ...entry, matchPercent };
  });

  return scored
    .filter(e => e.matchPercent > 0)
    .sort((a, b) => b.matchPercent - a.matchPercent)
    .slice(0, limit);
}

/**
 * Hybrid search: 80% vector similarity + 20% keyword bonus.
 * Falls back to keyword-only if sqlite-vec unavailable.
 */
async function hybridSearch(conn, query, embedding, { limit = 5, category } = {}) {
  if (!query || query.trim() === '') {
    return db.listEntries(conn, { category, limit, recent: true })
      .map(e => ({ ...e, matchPercent: 100 }));
  }

  if (!embedding || !db.hasVec(conn)) {
    return keywordSearch(conn, query, { limit, category });
  }

  // Vector search: get 2x candidates
  const candidateLimit = limit * 2;
  let vecResults;
  try {
    if (category) {
      vecResults = conn.prepare(`
        SELECT ev.id, ev.distance, e.*
        FROM entries_vec ev
        JOIN entries e ON e.id = ev.id
        WHERE ev.embedding MATCH ? AND e.category = ?
        ORDER BY ev.distance
        LIMIT ?
      `).all(embedding, category, candidateLimit);
    } else {
      vecResults = conn.prepare(`
        SELECT ev.id, ev.distance, e.*
        FROM entries_vec ev
        JOIN entries e ON e.id = ev.id
        WHERE ev.embedding MATCH ?
        ORDER BY ev.distance
        LIMIT ?
      `).all(embedding, candidateLimit);
    }
  } catch {
    return keywordSearch(conn, query, { limit, category });
  }

  // Score: 80% vector + 20% keyword
  const scored = vecResults.map(row => {
    const vectorScore = 1 - row.distance; // cosine distance → similarity
    const keywordBonus = computeKeywordBonus(query, row.content + ' ' + (row.tags || ''));
    const finalScore = vectorScore * 0.8 + keywordBonus;
    const matchPercent = Math.round(Math.min(Math.max(finalScore * 100, 0), 100));
    return { ...row, matchPercent, distance: undefined };
  });

  return scored
    .filter(e => e.matchPercent > 20) // Filter noise below 20%
    .sort((a, b) => b.matchPercent - a.matchPercent)
    .slice(0, limit);
}

/**
 * Check for near-duplicate entries (cosine > 0.92).
 * Returns existing entry if duplicate found, null otherwise.
 */
function checkDuplicate(conn, content, embedding) {
  if (!embedding || !db.hasVec(conn)) {
    // Keyword fallback: exact content match
    const exact = conn.prepare('SELECT * FROM entries WHERE content = ?').get(content);
    if (exact) return exact;
    const exactPending = conn.prepare("SELECT * FROM pending WHERE content = ? AND status = 'pending'").get(content);
    return exactPending || null;
  }

  // Vector search for near-duplicates
  try {
    const result = conn.prepare(`
      SELECT ev.id, ev.distance
      FROM entries_vec ev
      WHERE ev.embedding MATCH ?
      ORDER BY ev.distance
      LIMIT 1
    `).get(embedding);

    if (result && (1 - result.distance) > 0.92) {
      return db.getEntry(conn, result.id);
    }

    // Also check pending
    const pendingResult = conn.prepare(`
      SELECT pv.id, pv.distance
      FROM pending_vec pv
      WHERE pv.embedding MATCH ?
      ORDER BY pv.distance
      LIMIT 1
    `).get(embedding);

    if (pendingResult && (1 - pendingResult.distance) > 0.92) {
      return db.getPending(conn, pendingResult.id);
    }
  } catch {
    // Vec not available
  }

  return null;
}

module.exports = { hybridSearch, keywordSearch, computeKeywordBonus, checkDuplicate };
```

```js
'use strict';

const db = require('./db');
const search = require('./search');

async function getEmbed() {
  if (process.env.WF_BRAIN_SKIP_EMBED === '1') return null;
  try {
    const embed = require('./embed');
    return embed;
  } catch {
    return null;
  }
}

function parseArgs(argv) {
  const command = argv[2];
  const args = { _positional: [] };
  for (let i = 3; i < argv.length; i++) {
    if (argv[i].startsWith('--')) {
      const key = argv[i].slice(2);
      // Flags that take no value
      if (key === 'force' || key === 'pretty' || key === 'recent' || key === 'approve-all') {
        args[key] = true;
      } else if (i + 1 < argv.length && !argv[i + 1].startsWith('--')) {
        args[key] = argv[i + 1];
        i++;
      } else {
        args[key] = true;
      }
    } else {
      args._positional.push(argv[i]);
    }
  }
  return { command, args };
}

function output(data, pretty) {
  if (pretty) {
    prettyPrint(data);
  } else {
    console.log(JSON.stringify(data));
  }
}

function prettyPrint(data) {
  if (Array.isArray(data)) {
    if (data.length === 0) {
      console.log('No results.');
      return;
    }
    data.forEach((entry, i) => {
      const match = entry.matchPercent != null ? ` (${entry.matchPercent}% match)` : '';
      console.log(`\n[${i + 1}]${match} ${entry.category} — ${entry.created_at || ''}`);
      console.log(`    ${entry.content}`);
      if (entry.tags) console.log(`    Tags: ${entry.tags}`);
    });
  } else {
    console.log(JSON.stringify(data, null, 2));
  }
}

async function run(argv) {
  const { command, args } = parseArgs(argv);
  const pretty = args.pretty;

  if (!command || command === 'help') {
    output({ commands: ['init', 'search', 'store', 'propose', 'review', 'list', 'edit', 'delete', 'stats'] });
    return;
  }

  // For init, use project-dir flag or cwd
  if (command === 'init') {
    const projectDir = args['project-dir'] || process.cwd();
    const dbPath = require('path').join(projectDir, '.claude', 'brain.db');
    const conn = db.initDb(dbPath);
    const stats = db.getStats(conn);
    conn.close();
    output({ success: true, dbPath, ...stats }, pretty);
    return;
  }

  // All other commands need an existing brain
  const dbPath = db.findBrainDb(process.cwd());
  if (!dbPath) {
    output({ error: 'No brain.db found. Run: wf-brain init' });
    process.exitCode = 1;
    return;
  }

  const conn = db.initDb(dbPath);
  const embed = await getEmbed();

  try {
    switch (command) {
      case 'store': {
        const content = args._positional.join(' ');
        if (!content) { output({ error: 'Content required' }); return; }
        if (!args.category) { output({ error: '--category required' }); return; }

        const embedding = embed ? await embed.generateEmbedding(content) : null;

        if (!args.force) {
          const dup = search.checkDuplicate(conn, content, embedding);
          if (dup) {
            output({ error: `Similar entry already exists (id: ${dup.id})`, duplicate: dup.id });
            return;
          }
        }

        const id = db.insertEntry(conn, {
          content,
          category: args.category,
          tags: args.tags || '',
          source: args.source || 'manual',
          embedding
        });
        output({ id, stored: true }, pretty);
        break;
      }

      case 'propose': {
        const content = args._positional.join(' ');
        if (!content) { output({ error: 'Content required' }); return; }
        if (!args.category) { output({ error: '--category required' }); return; }

        const embedding = embed ? await embed.generateEmbedding(content) : null;

        if (!args.force) {
          const dup = search.checkDuplicate(conn, content, embedding);
          if (dup) {
            output({ error: `Similar entry already exists (id: ${dup.id})`, duplicate: dup.id });
            return;
          }
        }

        const id = db.insertPending(conn, {
          content,
          category: args.category,
          tags: args.tags || '',
          source: args.source || '',
          proposedBy: args.source || '',
          embedding
        });
        output({ id, proposed: true }, pretty);
        break;
      }

      case 'search': {
        const query = args._positional.join(' ');
        const embedding = embed ? await embed.generateEmbedding(query) : null;
        const results = await search.hybridSearch(conn, query, embedding, {
          limit: parseInt(args.limit) || 5,
          category: args.category
        });
        output(results, pretty);
        break;
      }

      case 'review': {
        if (args.approve) {
          const id = parseInt(args.approve);
          const entryId = db.approvePending(conn, id);
          output({ approved: id, entryId }, pretty);
        } else if (args.reject) {
          const id = parseInt(args.reject);
          db.rejectPending(conn, id);
          output({ rejected: id, status: 'rejected' }, pretty);
        } else if (args['approve-all']) {
          const ids = db.approveAllPending(conn);
          output({ approved: ids.length, entryIds: ids }, pretty);
        } else {
          const pending = db.listPending(conn);
          output(pending, pretty);
        }
        break;
      }

      case 'list': {
        const results = db.listEntries(conn, {
          category: args.category,
          limit: parseInt(args.limit) || 20,
          recent: args.recent
        });
        output(results, pretty);
        break;
      }

      case 'edit': {
        const id = parseInt(args._positional[0]);
        if (!id) { output({ error: 'Entry ID required' }); return; }
        const fields = {};
        if (args.content) fields.content = args.content;
        if (args.category) fields.category = args.category;
        if (args.tags) fields.tags = args.tags;
        if (Object.keys(fields).length === 0) { output({ error: 'No fields to update' }); return; }
        // Regenerate embedding if content changed
        if (fields.content && embed) {
          const newEmbed = await embed.generateEmbedding(fields.content);
          if (newEmbed) fields.embedding = newEmbed;
        }
        db.updateEntry(conn, id, fields);
        output({ success: true, id }, pretty);
        break;
      }

      case 'delete': {
        const id = parseInt(args._positional[0]);
        if (!id) { output({ error: 'Entry ID required' }); return; }
        db.deleteEntry(conn, id);
        output({ success: true, id }, pretty);
        break;
      }

      case 'stats': {
        const stats = db.getStats(conn);
        output(stats, pretty);
        break;
      }

      default:
        output({ error: `Unknown command: ${command}` });
    }
  } finally {
    conn.close();
  }
}

module.exports = { run, parseArgs };
```

```js
'use strict';

const DECISION_PATTERNS = [
  /\bdecision\b/i,
  /\bchose\b/i,
  /\bbecause\b/i,
  /\binstead of\b/i,
  /\bswitched to\b/i,
  /\bwent with\b/i,
  /\bpicked\b/i,
  /\bover\b.*\bfor\b/i
];

/**
 * Parse standards.md into seed entries.
 * Splits on ## headings, filters by length, caps at 20.
 */
function parseStandards(content) {
  const sections = content.split(/^## /m).slice(1); // Skip content before first ##
  const entries = [];

  for (const section of sections) {
    if (entries.length >= 20) break;

    const lines = section.split('\n');
    const heading = lines[0].trim();
    const body = lines.slice(1).join('\n').trim();

    if (body.length < 50 || body.length > 500) continue;

    const tags = heading.toLowerCase().replace(/[^a-z0-9\s]/g, '').trim().split(/\s+/).join(',');

    entries.push({
      content: body,
      category: 'convention',
      tags,
      source: 'seed:standards.md'
    });
  }

  return entries;
}

/**
 * Parse progress.md for decision-like paragraphs.
 * Looks for keywords like "decision", "chose", "because", "instead of".
 * Caps at 10 entries.
 */
function parseProgress(content) {
  const entries = [];
  const paragraphs = content.split(/\n(?=##\s)/);

  for (const para of paragraphs) {
    if (entries.length >= 10) break;

    // Check if paragraph contains decision-like patterns
    const hasDecision = DECISION_PATTERNS.some(p => p.test(para));
    if (!hasDecision) continue;

    // Extract the relevant lines (heading + decision content)
    const lines = para.split('\n').filter(l => l.trim());
    const heading = lines[0] || '';

    // Find the decision-specific line(s)
    const decisionLines = lines.filter(l =>
      DECISION_PATTERNS.some(p => p.test(l))
    );

    if (decisionLines.length === 0) continue;

    const decisionContent = decisionLines.join(' ').replace(/\*\*/g, '').trim();
    if (decisionContent.length < 20 || decisionContent.length > 500) continue;

    // Determine category based on keywords
    const category = /\bswitch/i.test(decisionContent) || /\binstead of/i.test(decisionContent)
      ? 'decision' : 'history';

    const tags = heading.replace(/^#+\s*/, '').toLowerCase()
      .replace(/[^a-z0-9\s]/g, '').trim().split(/\s+/).slice(0, 3).join(',');

    entries.push({
      content: decisionContent,
      category,
      tags: tags || 'progress',
      source: 'seed:progress.md'
    });
  }

  return entries;
}

module.exports = { parseStandards, parseProgress };
```

```js
#!/usr/bin/env node
'use strict';

const { execFileSync, execSync } = require('child_process');
const path = require('path');
const fs = require('fs');

// Auto-install deps on first run
const nmPath = path.join(__dirname, 'node_modules');
if (!fs.existsSync(nmPath)) {
  execSync('npm install --production', { cwd: __dirname, stdio: 'pipe' });
}

const { McpServer } = require('@modelcontextprotocol/sdk/server/mcp.js');
const { StdioServerTransport } = require('@modelcontextprotocol/sdk/server/stdio.js');
const { z } = require('zod');

// Resolve CLI path relative to wf-system root
// MCP server is at: {wf-system}/.claude/mcp-servers/wf-brain/index.js
// CLI is at: {wf-system}/scripts/wf-brain.js
const WF_SYSTEM_ROOT = path.resolve(__dirname, '..', '..', '..');
const CLI_PATH = path.join(WF_SYSTEM_ROOT, 'scripts', 'wf-brain.js');

function runCli(args, cwd) {
  try {
    const result = execFileSync('node', [CLI_PATH, ...args], {
      cwd: cwd || process.cwd(),
      encoding: 'utf8',
      timeout: 30000,
      env: { ...process.env }
    });
    return JSON.parse(result.trim());
  } catch (err) {
    const stderr = err.stderr || '';
    const stdout = err.stdout || '';
    try {
      return JSON.parse(stdout.trim());
    } catch {
      return { error: stderr || err.message };
    }
  }
}

const server = new McpServer({
  name: 'wf-brain',
  version: '1.0.0'
});

server.tool(
  'brain_search',
  'Search the project knowledge brain for relevant information',
  {
    query: z.string().describe('Search query'),
    limit: z.number().optional().default(5).describe('Max results'),
    category: z.string().optional().describe('Filter by category')
  },
  async ({ query, limit, category }) => {
    const args = ['search', query, '--limit', String(limit)];
    if (category) args.push('--category', category);
    const result = runCli(args);
    return { content: [{ type: 'text', text: JSON.stringify(result, null, 2) }] };
  }
);

server.tool(
  'brain_store',
  'Store approved knowledge in the project brain (main agent only)',
  {
    content: z.string().describe('Knowledge content (2-4 sentences)'),
    category: z.enum(['architecture', 'domain', 'convention', 'gotcha', 'decision', 'history']),
    tags: z.string().optional().describe('Comma-separated tags'),
    source: z.string().optional().default('manual')
  },
  async ({ content, category, tags, source }) => {
    const args = ['store', '--category', category, '--source', source, content];
    if (tags) args.splice(3, 0, '--tags', tags);
    const result = runCli(args);
    return { content: [{ type: 'text', text: JSON.stringify(result, null, 2) }] };
  }
);

server.tool(
  'brain_propose',
  'Propose knowledge for review (sub-agents use this instead of store)',
  {
    content: z.string().describe('Knowledge content (2-4 sentences)'),
    category: z.enum(['architecture', 'domain', 'convention', 'gotcha', 'decision', 'history']),
    tags: z.string().optional().describe('Comma-separated tags'),
    source: z.string().optional().default('agent:unknown')
  },
  async ({ content, category, tags, source }) => {
    const args = ['propose', '--category', category, '--source', source, content];
    if (tags) args.splice(3, 0, '--tags', tags);
    const result = runCli(args);
    return { content: [{ type: 'text', text: JSON.stringify(result, null, 2) }] };
  }
);

server.tool(
  'brain_review',
  'List, approve, or reject pending knowledge entries',
  {
    action: z.enum(['list', 'approve', 'reject', 'approve-all']).default('list'),
    id: z.number().optional().describe('Pending entry ID (for approve/reject)')
  },
  async ({ action, id }) => {
    const args = ['review'];
    if (action === 'approve' && id) args.push('--approve', String(id));
    else if (action === 'reject' && id) args.push('--reject', String(id));
    else if (action === 'approve-all') args.push('--approve-all');
    const result = runCli(args);
    return { content: [{ type: 'text', text: JSON.stringify(result, null, 2) }] };
  }
);

server.tool(
  'brain_stats',
  'Get brain statistics: entry count by category, pending count',
  {},
  async () => {
    const result = runCli(['stats']);
    return { content: [{ type: 'text', text: JSON.stringify(result, null, 2) }] };
  }
);

async function main() {
  const transport = new StdioServerTransport();
  await server.connect(transport);
}

main().catch(err => {
  console.error('wf-brain MCP server error:', err);
  process.exit(1);
});
```

```python
    # -------------------------------------------------------------------------
    # Brain Integration
    # -------------------------------------------------------------------------

    def _brain_search(self, keywords: str, limit: int = 5) -> Optional[str]:
        """Search the brain for relevant knowledge."""
        if not keywords or not keywords.strip():
            return None
        try:
            # Resolve wf-brain CLI path
            hooks_dir = Path(__file__).parent
            wf_system_root = hooks_dir.parent
            cli_path = wf_system_root / "scripts" / "wf-brain.js"

            if not cli_path.exists():
                return None

            result = subprocess.run(
                ["node", str(cli_path), "search", keywords, "--limit", str(limit)],
                capture_output=True, text=True, timeout=10, cwd=self.cwd
            )
            if result.returncode != 0:
                return None

            entries = json.loads(result.stdout.strip())
            if not entries or not isinstance(entries, list):
                return None

            lines = ["Brain Context (auto-retrieved):"]
            for entry in entries:
                cat = entry.get("category", "")
                content = entry.get("content", "")
                match_pct = entry.get("matchPercent", "")
                lines.append(f"- [{cat}] ({match_pct}% match) {content}")

            return "\n".join(lines)
        except (subprocess.TimeoutExpired, FileNotFoundError, json.JSONDecodeError, Exception):
            return None

    def _brain_stats(self) -> Optional[Dict[str, Any]]:
        """Get brain stats (entry count, pending count)."""
        try:
            hooks_dir = Path(__file__).parent
            wf_system_root = hooks_dir.parent
            cli_path = wf_system_root / "scripts" / "wf-brain.js"

            if not cli_path.exists():
                return None

            result = subprocess.run(
                ["node", str(cli_path), "stats"],
                capture_output=True, text=True, timeout=10, cwd=self.cwd
            )
            if result.returncode != 0:
                return None

            return json.loads(result.stdout.strip())
        except (subprocess.TimeoutExpired, FileNotFoundError, json.JSONDecodeError, Exception):
            return None
```
